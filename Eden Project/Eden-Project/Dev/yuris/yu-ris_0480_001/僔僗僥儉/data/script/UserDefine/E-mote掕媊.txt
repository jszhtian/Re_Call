#=ES.EMOTE.DEFINE


	//---------------------------------------------------------------------
	// 以下で髪、胸、パーツの揺れ具合を設定することができます。
	// 最終的な揺れ具合は 0 を最小、256 を最大として
	//  (揺れスケール最大値) × (揺れスケール【コンフィグ指定値】) ÷ 256
	// となります。
	//---------------------------------------------------------------------

	//---------------------------------------------------------------------
	// 髪揺れスケール最大値
	//---------------------------------------------------------------------
		@T(51,01) = 128   // 0〜256

	//---------------------------------------------------------------------
	// 胸揺れスケール最大値
	//---------------------------------------------------------------------
		@T(51,02) = 128   // 0〜256

	//---------------------------------------------------------------------
	// パーツ揺れスケール最大値
	//---------------------------------------------------------------------
		@T(51,03) = 128   // 0〜256


	//---------------------------------------------------------------------
	// 髪揺れスケール【コンフィグ指定値】(初回起動時にセットされる)
	//---------------------------------------------------------------------
		@T(51,11) = 256   // 0〜256

	//---------------------------------------------------------------------
	// 胸揺れスケール【コンフィグ指定値】(初回起動時にセットされる)
	//---------------------------------------------------------------------
		@T(51,12) = 256   // 0〜256

	//---------------------------------------------------------------------
	// パーツ揺れスケール【コンフィグ指定値】(初回起動時にセットされる)
	//---------------------------------------------------------------------
		@T(51,13) = 256   // 0〜256



RETURN[]
