#=ES.CGCXY.DEF


	//----------------------------------------------------------------
	// 「背景中心点」と「背景を表示した際のカメラ座標」の定義
	//
	//    \BGCXY.DEF( 背景ファイル名, 背景中心点X, 背景中心点Y, 背景距離Z, カメラX, カメラY, カメラZ )
	//
	//  ※定義を省略した場合は、
	//    画像の幅と高さの中心がそのまま中心点に、またカメラの位置は XYZ=(0,0,0) になります。
	//
	//----------------------------------------------------------------
	\BGCXY.DEF("white" , 400, 300, 200, 0, 0, 0)	//白
	\BGCXY.DEF("black" , 400, 300, 200, 0, 0, 0)	//黒
//	\BGCXY.DEF("BG_010", 400, 300, 200, 0, 0, 0)	//


	//----------------------------------------------------------------
	//
	// \CGCXY.DEF( 画像ファイル名 , 中心点Ｘ , 中心点Ｙ )
	//
	//  ※定義を省略した場合は、
	//    画像の幅と高さの中心がそのまま中心点になります。
	//
	//----------------------------------------------------------------

	//----------------------------------------------------------------
	//■立ち絵
	//----------------------------------------------------------------

	//============================================ 優理
	\CGCXY.DEF("yuu_1a", 196, 180)
	\CGCXY.DEF("yuu_1b", 196, 180)
	\CGCXY.DEF("yuu_2a", 196, 180)
	\CGCXY.DEF("yuu_2b", 196, 180)


	//============================================ 枝理
	\CGCXY.DEF("eri_1a", 234, 190)
	\CGCXY.DEF("eri_1b", 234, 190)
	\CGCXY.DEF("eri_2a", 234, 190)
	\CGCXY.DEF("eri_2b", 234, 190)


	//============================================ きゅーたろー
	\CGCXY.DEF("qut_1a", 330, 315)
	\CGCXY.DEF("qut_1b", 330, 315)
	\CGCXY.DEF("qut_2a", 330, 315)
	\CGCXY.DEF("qut_2b", 330, 315)
