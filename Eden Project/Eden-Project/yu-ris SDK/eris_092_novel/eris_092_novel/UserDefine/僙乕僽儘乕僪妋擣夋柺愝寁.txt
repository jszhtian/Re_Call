//・ERIS ver0.72  001→002 変更点

//      (ゲーム中以外からの)ロード確認の有無を設定できるようになった

\VER.SAVELOADCONFDEF(002) //定義Version

//-------------------------------------------------------------------------
//
// 全般
//
//  ※(未)と記している部分は未実装の項目です。
//
//-------------------------------------------------------------------------
#=es.SceneSaveLoadConf.Define

	//---------------------------------------------------------------------
	// 確認設定
	//---------------------------------------------------------------------
		\SAVE.CONF(0)       //セーブ確認の有無(1=有/0=無)
		\SAVE.CONF.MODE(1)  //セーブ確認に画像を使う(=1)／ダイアログを使う(=0)
		\SAVE.CONF.WRITE(0) //セーブ確認をセーブデータ上書き時のみにする(1=ON/0=OFF)
		                    //※ONにする場合、画像やメッセージの文章を「上書きセーブしますか？」に、また
		                    //オプション設定の「セーブ確認」の項目も「上書きセーブ確認」に変更する事を強く推奨します
		\LOAD.CONF(1, 0)    //(ゲーム中からの)ロード確認の有無(1=有/0=無), (ゲーム中以外からの)ロード確認の有無(1=有/0=無)
		                    //※第１パラメータが 0(=無) の場合、第２パラメータは無視されます(自動的に0(=無)になります)
		                    //  また、オプション画面で値を変更できるのは第１パラメータのほうになります。
		\LOAD.CONF.MODE(1)  //ロード確認に画像を使う(=1)／ダイアログを使う(=0)

		\SAVE.CONF.TX("セーブします。よろしいですか？") //セーブ確認がダイアログの場合のメッセージ
		\LOAD.CONF.TX("ロードします。よろしいですか？") //ロード確認がダイアログの場合のメッセージ

		\SAVE.CONF.END(0)                               //セーブ後メッセージの有無(1=有/0=無)
		\SAVE.CONF.END.TX("セーブしました。")           //セーブ後メッセージ

		\SL.COPY.CONF(1)      //セーブデータコピー確認
		\SL.COPY.CONF.MODE(1) //コピー確認に画像を使う(=1)／ダイアログを使う(=0)
		\SL.DEL.CONF(1)       //セーブデータ削除確認
		\SL.DEL.CONF.MODE(1)  //削除確認に画像を使う(=1)／ダイアログを使う(=0)

		\SL.COPY.CONF.TX("コピーします。宜しいですか？") //コピー確認がダイアログの場合のメッセージ
		\SL.DEL.CONF.TX("削除します。宜しいですか？")    //削除確認がダイアログの場合のメッセージ

	//---------------------------------------------------------------------
	// どの画像にどの機能を割り振るかの設定
	// 　ここで定義したボタン番号の画像が、その機能となります。
	//   番号が大きいほうが前面に表示されます。
	//---------------------------------------------------------------------
		@es.CNUM.SAVECONF.YES   = 11 //セーブ確認・はい
		@es.CNUM.SAVECONF.NO    = 12 //セーブ確認・いいえ
		@es.CNUM.LOADCONF.YES   = 11 //ロード確認・はい
		@es.CNUM.LOADCONF.NO    = 12 //ロード確認・いいえ
		@es.CNUM.SLCOPYCONF.YES = 11 //コピー確認・はい
		@es.CNUM.SLCOPYCONF.NO  = 12 //コピー確認・いいえ
		@es.CNUM.SLDELCONF.YES  = 11 //削除確認・はい
		@es.CNUM.SLDELCONF.NO   = 12 //削除確認・いいえ


Return[]
