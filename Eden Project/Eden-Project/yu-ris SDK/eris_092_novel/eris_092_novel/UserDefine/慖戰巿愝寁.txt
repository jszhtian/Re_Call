#=es.SceneSelect.Define


	//---------------------------------------------------------------------
	// 選択肢
	//---------------------------------------------------------------------

	//画像設定
		\SEL.CG.DEF("cg/select/tip_select_off", "cg/select/tip_select_over") // 選択肢画像
		\SEL.XY.DEF(99,200)           // 選択肢画像の基準座標
		\SEL.ALIGN.Y(1)               // 選択肢画像の縦方向の表示方法(0=上揃え／1=中央揃え)

	//効果音設定
		\SEL.SE("", "")               // オンマウス時ＳＥ, クリック時ＳＥ

	//選択肢設定
		//\SEL.TYPE(0)                // (未)選択肢の形式(画像形式/文字反転形式)
		\SEL.LINESPACE(70)            // 選択肢と選択肢の間隔(※注：選択肢画像自体の縦幅も含める)
		\SEL.TX.Y(17)                 // 選択肢文字の相対表示Ｙ座標
		\SEL.TX.SIZE(24)              // 選択肢文字サイズ
		\SEL.TX.COLOR(-1)             // 選択肢文字色:通常(RGB指定)(-1の場合はテキストの標準色に従う)
		\SEL.TX.AL.COLOR(0x88eeff)    // 選択肢文字色:選択済(RGB指定)(-1の場合はテキストの標準色に従う)
		\SEL.TX.SHADE(1)              // 影を付ける(=1)／付けない(=0)
		\SEL.TX.SHADE.COLOR(-1)       // 選択肢文字影色:通常(RGB指定)(-1の場合はテキスト影の色に従う)
		\SEL.TX.AL.SHADE.COLOR(-1)    // 選択肢文字影色:選択済(RGB指定)(-1の場合はテキストの標準色に従う)
		\SEL.TX.SHADE.XY(1,1)         // 影を付ける場合の影の位置座標（相対座標指定）
		                              // （例：2,1 なら右に２ピクセル、下に１ピクセルの意）


		\SEL.NA.A(128)                // 選択項目が選択不可の時の、選択肢バー画像の不透明度(0〜256)
		\SEL.TX.NA.A(256)             // 選択項目が選択不可の時の、選択肢文字の不透明度(0〜256)
		\SEL.TX.NA.COLOR(0xa0a0a0)    // 選択項目が選択不可の時の、選択肢文字色(RGB指定)(-1の場合は \SEL.TX.COLOR の色に従う)
		\SEL.TX.NA.SHADE(1)           // 影を付ける(=1)／付けない(=0)
		\SEL.TX.NA.SHADE.COLOR(-1)    // 選択項目が選択不可の時の、影の色(RGB指定)(-1の場合は \SEL.TX.SHADE.COLOR の色に従う)
		\SEL.TX.NA.SHADE.XY(1,1)      // 影を付ける場合の影の位置座標（相対座標指定）


		\SEL.LOG(1)                   // 選択結果をバックログに記録する
		\SEL.AUTOMODE.DEF(1)          // 選択肢を通過後もオートモードを持続する(=1)／しない(=0)  ※(初回起動時にセットされる)
		\SEL.SKIPMODE.DEF(1)          // 選択肢を通過後もスキップモードを持続する(=1)／しない(=0)  ※(初回起動時にセットされる)



return[]
