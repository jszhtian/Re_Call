//============================================================================
//
// ERIS 立ち絵定義
//
//============================================================================
\VER.STDEF(001) //定義Version
//============================================================================

#=立ち絵定義


	//---------------------------------------------------------------------
	// 立ち絵画像のあるパス
	//---------------------------------------------------------------------
		\ST.PATH("cg/stand/")


	//---------------------------------------------------------------------
	// 立ち絵のデフォルト切替時間(ミリ秒)
	//---------------------------------------------------------------------
		\ST.FADE.DEF(250)


	//---------------------------------------------------------------------
	// 立ち絵切替時にメッセージウィンドウを消去するか否か
	//---------------------------------------------------------------------
		\ST.TX.HIDE.DEF(0)  //(1=消去する/0=しない)


	//---------------------------------------------------------------------
	// 立ち絵切替時にボタンチップを消去するか否か
	//---------------------------------------------------------------------
		\ST.TIP.HIDE.DEF(0)  //(1=消去する/0=しない)


	//---------------------------------------------------------------------
	// キャラ毎にフォルダを分けるか否か(1=する/0=しない)
	//---------------------------------------------------------------------
		\ST.PATH.CHARAFLAG(1)


	//---------------------------------------------------------------------
	// キャラ毎にフォルダを分ける際のフォルダ名
	//
	//                 ↓※この番号は macro.txt ファイルで定義した
	//                 ↓  「立ち絵マクロ定義」のキャラ番号と
	//                 ↓  合わせてください。
	//---------------------------------------------------------------------
	    \ST.PATH.CHARA(01, "yuuri/")
	    \ST.PATH.CHARA(02, "eri/")
	    \ST.PATH.CHARA(03, "ryo/")
	    \ST.PATH.CHARA(04, "hiro/")


	//---------------------------------------------------------------------
	// キャラ毎の基準表示位置  (キャラ番号, Ｘ座標, Ｙ座標)
	// ※立ち絵画像が中心に表示されるように調整します。
	//
	//               ↓※この番号は macro.txt ファイルで定義した
	//               ↓  「立ち絵マクロ定義」のキャラ番号と
	//               ↓  合わせてください。
	//---------------------------------------------------------------------
	    \ST.CHARA.XY(01, 240, 044) //01:優理
	    \ST.CHARA.XY(02, 200, 104) //02:枝理
	    \ST.CHARA.XY(03, 000, 000) //03:良介
	    \ST.CHARA.XY(04, 000, 000) //04:広伸


	//---------------------------------------------------------------------
	// 立ち位置簡易利用文字列定義
	//
	// [例]
	//  \ST.POS.X(50); \ST.POSDEF("やや右")
	// と定義すると、
	// \枝理(1003,やや右) としたときに、
	// 立ち絵が、基準表示位置＋５０の位置に表示されます。
	//
	//---------------------------------------------------------------------
		\ST.POS.X(-200      ); \ST.POSDEF("L")
		\ST.POS.X(   0      ); \ST.POSDEF("M")
		\ST.POS.X( 200      ); \ST.POSDEF("R")
		\ST.POS.X(-200-66-22); \ST.POSDEF("P1")
		\ST.POS.X(-200-66   ); \ST.POSDEF("P2")
		\ST.POS.X(-200-66+22); \ST.POSDEF("P3")
		\ST.POS.X(-200   -22); \ST.POSDEF("P4")
		\ST.POS.X(-200      ); \ST.POSDEF("P5")
		\ST.POS.X(-200   +22); \ST.POSDEF("P6")
		\ST.POS.X(-200+66-22); \ST.POSDEF("P7")
		\ST.POS.X(-200+66   ); \ST.POSDEF("P8")
		\ST.POS.X(-200+66+22); \ST.POSDEF("P9")
		\ST.POS.X(   0-66-22); \ST.POSDEF("P10")
		\ST.POS.X(   0-66   ); \ST.POSDEF("P11")
		\ST.POS.X(   0-66+22); \ST.POSDEF("P12")
		\ST.POS.X(   0   -22); \ST.POSDEF("P13")
		\ST.POS.X(   0      ); \ST.POSDEF("P14")
		\ST.POS.X(   0   +22); \ST.POSDEF("P15")
		\ST.POS.X(   0+66-22); \ST.POSDEF("P16")
		\ST.POS.X(   0+66   ); \ST.POSDEF("P17")
		\ST.POS.X(   0+66+22); \ST.POSDEF("P18")
		\ST.POS.X( 200-66-22); \ST.POSDEF("P19")
		\ST.POS.X( 200-66   ); \ST.POSDEF("P20")
		\ST.POS.X( 200-66+22); \ST.POSDEF("P21")
		\ST.POS.X( 200   -22); \ST.POSDEF("P22")
		\ST.POS.X( 200      ); \ST.POSDEF("P23")
		\ST.POS.X( 200   +22); \ST.POSDEF("P24")
		\ST.POS.X( 200+66-22); \ST.POSDEF("P25")
		\ST.POS.X( 200+66   ); \ST.POSDEF("P26")
		\ST.POS.X( 200+66+22); \ST.POSDEF("P27")


	//---------------------------------------------------------------------
	// 複数キャラ同時表示時の定義文字列 (※↑で定義した文字列で指定する)
	//---------------------------------------------------------------------
		\ST.POS.X.SYNC.2("L", "R")      //２人同時
		\ST.POS.X.SYNC.3("L", "M", "R") //３人同時



Return[]
