#=効果音定義


	//---------------------------------------------------------------------
	// 効果音素材のあるパス
	//---------------------------------------------------------------------
		\SE.PATH("se/")


	//---------------------------------------------------------------------
	// 効果音ボリューム初期値(初回起動時にセットされる)
	//---------------------------------------------------------------------
		\SE.VOL.DEF(204)  // 0〜256


	//---------------------------------------------------------------------
	// システム音ボリューム初期値(初回起動時にセットされる)
	//---------------------------------------------------------------------
		\SSE.VOL.DEF(230)  // 0〜256


	//---------------------------------------------------------------------
	// 全効果音のデフォルトフェード時間(ミリ秒)
	//---------------------------------------------------------------------
		\SE.FADE.DEF(2000)


	//---------------------------------------------------------------------
	// 効果音簡易利用文字列定義
	//---------------------------------------------------------------------

		//例：「Ａ０１」という文字列を定義する
		//
		// ここで「Ａ０１」という文字列で設定を定義しておくと、
		// スクリプト中に \SE("Ａ０１") という記述があった場合、
		// 以下に設定したパラメータが自動で設定された状態で再生されます。
		//
/*
		\SE.FILE("SE_A01")    // ファイル名
		\SE.FADE(0)           // フェード時間
		\SE.POINT(-1,-1)      // 再生開始／終了／再開始地点
		\SE.LOOP(1)           // 0=無限ループ再生, 1〜n=ループ回数
		\SE.WAIT(0)           // 1=再生終了待ち
		\SEDEF("A01")         // 登録.
*/

		//↓...現状100個分まで追加できます(β版仕様)



Return[]
