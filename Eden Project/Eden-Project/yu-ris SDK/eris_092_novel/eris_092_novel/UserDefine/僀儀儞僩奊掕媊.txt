\VER.EVDEF(002) //定義Version
//============================================================================

#=イベント絵定義


	//---------------------------------------------------------------------
	// イベント絵素材のあるパス
	//---------------------------------------------------------------------
		\EV.PATH("cg/ev/")


	//---------------------------------------------------------------------
	// イベント絵のデフォルト切替時間(ミリ秒)
	//---------------------------------------------------------------------
		\EV.FADE.DEF(1500)


	//---------------------------------------------------------------------
	// イベント絵切替時にメッセージウィンドウを消去するか否か
	//---------------------------------------------------------------------
		\EV.TX.HIDE.DEF(1)  //(1=消去する/0=しない)


	//---------------------------------------------------------------------
	// イベント絵切替時にボタンチップを消去するか否か
	//---------------------------------------------------------------------
		\EV.TIP.HIDE.DEF(1)  //(1=消去する/0=しない)


	//---------------------------------------------------------------------
	// イベント絵簡易利用文字列定義
	//---------------------------------------------------------------------

		//例：「出会い１」という文字列を定義する
		//
		// ここで「出会い１」という文字列で設定を定義しておくと、
		// スクリプト中に \EV("出会い１") という記述があった場合、
		// 以下に設定したパラメータが自動で設定された状態で読み込まれます。
		//
/*
		\EV.FILE("ev001_1")   // ファイル名
		\EV.FADE(1500)        // フェード時間
		\EV.TR("5")           // トランジション文字列
		\EVDEF("出会い１")    // 登録.
*/

		//↓...現状100個分まで追加できます(β版仕様)



Return[]
