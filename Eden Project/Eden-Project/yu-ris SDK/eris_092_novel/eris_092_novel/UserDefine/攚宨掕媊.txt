\VER.BGDEF(002) //定義Version
//============================================================================

#=背景定義


	//---------------------------------------------------------------------
	// 背景画像のあるパス
	//---------------------------------------------------------------------
		\BG.PATH("cg/bg/")


	//---------------------------------------------------------------------
	// 背景のデフォルト切替時間(ミリ秒)
	//---------------------------------------------------------------------
		\BG.FADE.DEF(750)


	//---------------------------------------------------------------------
	// 背景切替時にメッセージウィンドウを消去するか否か
	//---------------------------------------------------------------------
		\BG.TX.HIDE.DEF(1)  //(1=消去する/0=しない)


	//---------------------------------------------------------------------
	// 背景切替時にボタンチップを消去するか否か
	//---------------------------------------------------------------------
		\BG.TIP.HIDE.DEF(1)  //(1=消去する/0=しない)


	//---------------------------------------------------------------------
	// 背景簡易利用文字列定義
	//---------------------------------------------------------------------

		//例：「教室・昼」という文字列を定義する
		//
		// ここで「教室・昼」という文字列で設定を定義しておくと、
		// スクリプト中に \BG("教室・昼") という記述があった場合、
		// 以下に設定したパラメータが自動で設定された状態で読み込まれます。
		//
/*
		\BG.FILE("bg01_noon") // ファイル名
		\BG.FADE(1500)        // フェード時間
		\BG.TR("11")          // トランジション文字列
		\BGDEF("教室・昼")    // 登録.
*/

		//↓...現状100個分まで追加できます(β版仕様)



Return[]
