#=es.SceneSaveLoad.Define


	//---------------------------------------------------------------------
	// Newマーク設定
	//---------------------------------------------------------------------
		\SL.NEW.XY(-12, 62)              // 相対表示座標

	//---------------------------------------------------------------------
	// サムネイル画像設定
	//---------------------------------------------------------------------
		//\SL.THUMB(1)                   // (未)サムネイルをセーブする(=1)/しない(=0)
		\SL.THUMB.SIZE(100,75)           // サムネイルサイズ
		\SL.THUMB.XY(8,5)                // 相対表示座標

	//---------------------------------------------------------------------
	// テキスト設定
	//---------------------------------------------------------------------
		\SL.TL.SIZE(230,80)              // テキストレイヤサイズ
		\SL.TL.XY(115,6)                 // テキストレイヤ相対表示座標
		\SL.TX.FONT.DEF("ＭＳ ゴシック") // デフォルトフォント
		\SL.TX.SIZE(12)                  // テキストサイズ
		\SL.TX.COLOR(0xffffff)           // テキスト色
		\SL.TX.SPACE.Y(3)                // 文字の行間隔
		\SL.TX.SHADE(1)                  // 影を付ける(=1)／付けない(=0)
		\SL.TX.SHADE.COLOR(0x000000)     // 影の色(RGB指定)
		\SL.TX.SHADE.XY(1,1)             // 影を付ける場合の影の位置座標（相対座標指定）
		                                 // （例：2,1 なら右に２ピクセル、下に１ピクセルの意）

	//---------------------------------------------------------------------
	// テキスト内容設定
	//  \SL.PLATE.TX( 行 , 種類 )
	//
	//  種類
	//          0:なし
	//    1〜8191:$L.S(1〜8191)の内容
	// 9001〜9010:文章1〜10行目
	//       9011:現実の日付
	//       9012:ゲーム内の日付
	//---------------------------------------------------------------------
		\SL.PLATE.TX(1, 9011) //1行目 → 現実の日付
		\SL.PLATE.TX(2, 9001) //2行目 → 文章1行目
		\SL.PLATE.TX(3, 9002) //3行目 → 文章2行目
		\SL.PLATE.TX(4, 9003) //4行目 → 文章3行目
		\SL.PLATE.TX(5, 9004) //5行目 → 文章4行目

	//---------------------------------------------------------------------
	// 現実の日付、およびゲーム内の日付の表現方法
	//	\SL.DATE.TYPE( 年月日 , 時分秒 )
	//
	// 以下のパターンから選択できます。
	//
	// ※Y=年,M=月,D=日,W=曜日   ※H=時,M=分,S=秒
	// 年月日                    時分秒
	//  00:なし                  00:なし
	//  01:YYYY/MM/DD            01:HH:MM:SS
	//  02:YY/MM/DD              02:HH:MM
	//  03:YYYY/MM               03:MM:SS
	//  04:YY/MM                 04:HH
	//  05:MM/DD                 05:MM
	//  06:YYYY                  06:SS
	//  07:YY                    11:HH時MM分SS秒
	//  08:MM                    12:HH時MM分
	//  09:DD                    13:MM分SS秒
	//  10:YYYY/MM/DD(WW)        14:HH時
	//  11:YY/MM/DD(WW)          15:MM分
	//  12:MM/DD(WW)             16:SS秒
	//  13:DD(WW)
	//  21:YYYY年MM月DD日
	//  22:YY年MM月DD日
	//  23:YYYY年MM月
	//  24:YY年MM月
	//  25:MM月DD日
	//  26:YYYY年
	//  27:YY年
	//  28:MM月
	//  29:DD日
	//  30:YYYY年MM月DD日(WW)
	//  31:YY年MM月DD日(WW)
	//  32:MM月DD日(WW)
	//  33:DD日(WW)
	//---------------------------------------------------------------------
		\SL.PLATE.DATE(01, 01) //年月日, 時分秒


	//---------------------------------------------------------------------
	// サウンド設定
	//---------------------------------------------------------------------
		//\SL.SAVE.SE()                    // (未)セーブ時のＳＥ
		//\SL.LOAD.SE()                    // (未)ロード時のＳＥ
		//\SL.QSAVE.SE()                   // (未)クイックセーブ時のＳＥ
		//\SL.QLOAD.SE()                   // (未)クイックロード時のＳＥ

	//---------------------------------------------------------------------
	// 各種設定
	//---------------------------------------------------------------------
		\SL.PAGE.NUM(10)                   // ページ数(最大100まで)
		\SL.PLATE.NUM(10)                  // プレート数(最大30まで)

	//---------------------------------------------------------------------
	// どの画像にどの機能を割り振るかの設定
	// 　ここで定義したボタン番号の画像が、その機能となります。
	//   番号が大きいほうが前面に表示されます。
	//---------------------------------------------------------------------
		@es.CNUM.SL.PAGE   =  11 //ページ番号　※定義した番号＋ページ数 まで使われます(例:11なら11+ページ数(仮に8として)-1=18まで)
		@es.CNUM.SL.PLATE  = 111 //セーブプレート　※定義した番号＋プレート数 まで使われます(111なら111+プレート数(仮に12として)-1=122まで)
		@es.CNUM.SL.NEW    = 201 //New!マーク
		@es.CNUM.SL.COPY   = 251 //コピー
		@es.CNUM.SL.DELETE = 252 //削除
		@es.CNUM.SL.BACK   = 255 //戻る


Return[]
