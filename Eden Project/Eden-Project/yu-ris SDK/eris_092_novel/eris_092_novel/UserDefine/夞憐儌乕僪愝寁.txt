#=es.SceneReplayMode.Define

	//---------------------------------------------------------------------
	// 画像設定
	//---------------------------------------------------------------------
		\REPLAYMODE.THUMB.PATH("cg/ev_thumb/")      // サムネイル画像のあるパス
		\REPLAYMODE.NUM.FILE("cg/replaymode/num_")  //数字画像ファイル名(末尾に0〜9の数字が付いたファイル名を読みに行きます)
		\REPLAYMODE.SLH.FILE("cg/replaymode/slash") //スラッシュ画像ファイル名

	//---------------------------------------------------------------------
	// ＢＧＭ設定
	//---------------------------------------------------------------------
		\REPLAYMODE.BGM("")               // ＢＧＭファイル名orＢＧＭ定義文字列

	//---------------------------------------------------------------------
	// 各種設定
	//---------------------------------------------------------------------
		\REPLAYMODE.THUMB.NUM(8)          // １頁に表示するサムネイル枚数
		                                  // ここで指定した枚数分のプレートボタンを用意します。
		\REPLAYMODE.THUMB.XY(2,2)         // サムネイル相対表示座標(プレート座標を基準に)
		\REPLAYMODE.THUMB.SIZE(160,120)   // サムネイル画像サイズ
		\REPLAYMODE.THUMB.LIGHT(192,256)  // サムネイルの明るさ(通常,マウスオーバー時)(0〜256)
		\REPLAYMODE.CG.FADE(450,400,300)  // ＣＧフェードイン／クロスフェード／フェードアウト時間
		\REPLAYMODE.PAGE.FADE(450)        // (未)ページ切替フェード時間

		\REPLAYMODE.PAGENUM.XY(338,450)   // 現在のページ表示座標
		\REPLAYMODE.PAGENUM.SIZE(26)      // 数字１つ分の幅
		\REPLAYMODE.PERNUM.XY(310,510)    // 回想閲覧枚数表示座標
		\REPLAYMODE.PERNUM.SIZE(26)       // 数字１つ分の幅


	//---------------------------------------------------------------------
	// どの画像にどの機能を割り振るかの設定
	// 　ここで定義したボタン番号に対応する画像が、その機能となります。
	//   番号が大きいほうが前面に表示されます。
	//---------------------------------------------------------------------
		@es.CNUM.RPMODE.PLATE    = 101 //プレート(〜+ 上の\CGMODE.THUMB.NUMで定義した枚数分。デフォルトの場合だと8なので、つまり011〜018まで割り振られる事になる)
		@es.CNUM.RPMODE.PREVPAGE = 201 //前のページ
		@es.CNUM.RPMODE.NEXTPAGE = 202 //次のページ
		@es.CNUM.RPMODE.BACK     = 255 //戻る



Return[]

	// \BTN.KEY( ショートカットキー{一回の押下で一回のみ反応} ) 
	// \BTN.KEY.S( ショートカットキー{連続反応} ) 
	// \BTN.SE( オンマウス時ＳＥ, クリック時ＳＥ ) ※ファイル名orＳＥ定義文字列

	//---------------------------------------------------------------------
