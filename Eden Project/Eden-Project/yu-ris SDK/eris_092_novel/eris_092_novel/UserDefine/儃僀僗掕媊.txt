\VER.VODEF(004) //定義Version
//============================================================================

#=ボイス定義


	//---------------------------------------------------------------------
	// ボイス素材のあるパス
	//---------------------------------------------------------------------
		\VO.PATH("voice/")


	//---------------------------------------------------------------------
	// ボイス音量初期値(初回起動時にセットされる)
	//---------------------------------------------------------------------
		\VO.VOL.DEF(224)  // 0〜256


	//---------------------------------------------------------------------
	// ボイスハイライト時音量初期値(初回起動時にセットされる)
	//
	// ※ボイス再生時に、ＢＧＭや効果音の音量が自動的に下がる機能です。
	//   どのくらい下げるかを割合で設定します。
	//   256なら０割(＝ハイライト機能無効)で、128なら５割音量が下がります。
	//---------------------------------------------------------------------
		\VO.VOL.HIGH.DEF(176)  // 0〜256


	//---------------------------------------------------------------------
	// 自動フォルダ分け・文字列定義
	//
	//   ボイスファイル名の先頭に「文字列」が存在した場合、
	//   自動的に「ボイスフォルダ名」のフォルダ内を探しに行きます。
	//
	// [書き方]
	//   \VO.PATH.STR( 文字列 )
	//   \VO.PATH.FOLDER( ボイスフォルダ名 )
	//   \VO.PATHDEF( キャラ番号 )
	//
	// [例]
	//   \VO.PATH.STR("eri_")
	//   \VO.PATH.FOLDER("枝理/")
	//   \VO.PATHDEF(02)
	//
	//   と定義しておけば、
	//
	//   \VO(eri_0001)
	//
	//   とボイス命令があった場合、
	//   通常は「voice/eri_0001(.wav/.ogg)」ファイルを探しに行くところを、
	//   「voice/枝理/eri_0001(.wav/.ogg)」ファイルを探しに行くようになります。
	//
	//---------------------------------------------------------------------

	//■01:優理
		\VO.PATH.STR("yuuri_")     //ボイスファイル名先頭(※大文字小文字区別するので注意)
		\VO.PATH.FOLDER("優理/")   //ボイスフォルダ名
		\VO.PATHDEF(01)            //キャラ番号登録

	//■02:枝理
		\VO.PATH.STR("eri_")       //ボイスファイル名先頭(※大文字小文字区別するので注意)
		\VO.PATH.FOLDER("枝理/")   //ボイスフォルダ名
		\VO.PATHDEF(02)            //キャラ番号登録

		//↓...現状100個分まで追加できます(β版仕様)

	//■03:

	//■04:

	//■05:


	//---------------------------------------------------------------------
	// その他ボイスのキャラボイス番号設定
	//
	// 上で登録しなかったボイスファイル名の先頭が指定された場合、
	// 何番のキャラボイスとして音声再生するか設定します。
	//---------------------------------------------------------------------
		\VO.OTHER.PATH.FOLDER("")   //ボイスフォルダ名
		\VO.OTHER.PATHDEF(01)


Return[]
