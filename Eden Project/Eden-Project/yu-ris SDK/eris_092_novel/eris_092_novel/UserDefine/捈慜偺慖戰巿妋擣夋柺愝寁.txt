#=es.ScenePrevSelConf.Define

	//---------------------------------------------------------------------
	// 確認設定
	//---------------------------------------------------------------------
		\PREVSEL.CONF(1)       //確認する(=1)／確認しない(=0)
		\PREVSEL.CONF.MODE(1)  //確認に画像を使う(=1)／ダイアログを使う(=0)

		\PREVSEL.CONF.TX("直前の選択肢に戻ります。よろしいですか？") //ダイアログの場合のメッセージ


	//---------------------------------------------------------------------
	// どの画像にどの機能を割り振るかの設定
	// 　ここで定義したボタン番号の画像が、その機能となります。
	//   番号が大きいほうが前面に表示されます。
	//---------------------------------------------------------------------
		@es.CNUM.PREVSELCONF.YES = 11 //はい
		@es.CNUM.PREVSELCONF.NO  = 12 //いいえ


Return[]


//-------------------------------------------------------------------------
//
// 画像
//
//-------------------------------------------------------------------------
#=es.ScenePrevSelConf.BtnSet


	//---------------------------------------------------------------------
	//
	// ボタン
	//
	// \BTN.CG( ボタンOFF時の画像, マウスが乗った時の画像 )
	// \BTN.XY( 表示Ｘ座標, 表示Ｙ座標 )
	// \BTN.MAP.A( α値のしきい値(この値以上の画像部分がボタンとして反応します) )
	// \BTN.KEY( ショートカットキー{一回の押下で一回のみ反応} ) 
	// \BTN.KEY.S( ショートカットキー{連続反応} ) 
	// \BTN.SE( オンマウス時ＳＥ, クリック時ＳＥ ) ※ファイル名orＳＥ定義文字列
	//
	// \BTNDEF ← 最後にBTNDEFでボタンを登録します。
	//
	//---------------------------------------------------------------------

		//■背景
		\BTN.CG("cg/confirm/back")
		\BTN.XY(000, 000)
		\BTNDEF(01) //ボタン登録


		//■枠
		\BTN.CG("cg/confirm/dialog_prevsel")
		\BTN.XY(222, 214)
		\BTNDEF(10) //ボタン登録

		//■はい
		\BTN.CG("cg/confirm/btn_yes_off", "cg/confirm/btn_yes_over")
		\BTN.XY(293, 303)
		\BTN.KEY("ENTER")
		\BTNDEF(11) //ボタン登録

		//■いいえ
		\BTN.CG("cg/confirm/btn_no_off", "cg/confirm/btn_no_over")
		\BTN.XY(410, 303)
		\BTN.KEY("ESC", "BS", "MOUSE_R")
		\BTNDEF(12) //ボタン登録



Return[]
