	//---------------------------------------------------------------------
	// 画像設定
	//---------------------------------------------------------------------
		\CGMODE.THUMB.PATH("cg/ev_thumb/")    // サムネイル画像のあるパス
		\CGMODE.NUM.FILE("cg/cgmode/num_")    //数字画像ファイル名(末尾に0〜9の数字が付いたファイル名を読みに行きます)
		\CGMODE.PER.FILE("cg/cgmode/percent") //％画像ファイル名
		\CGMODE.SLH.FILE("cg/cgmode/slash")   //スラッシュ画像ファイル名

	//---------------------------------------------------------------------
	// ＢＧＭ設定
	//---------------------------------------------------------------------
		\CGMODE.BGM("")                // ＢＧＭファイル名orＢＧＭ定義文字列

	//---------------------------------------------------------------------
	// 各種設定
	//---------------------------------------------------------------------
		\CGMODE.THUMB.NUM(8)           // １頁に表示するサムネイル画像枚数
		                               // ここで指定した枚数分のプレートボタンを用意します。
		\CGMODE.THUMB.XY(2,2)          // サムネイル画像の相対表示座標(プレート座標を基準に)
		\CGMODE.THUMB.NUM.XY(112,100)  // サムネイル画像閲覧枚数の相対表示座標(プレート座標を基準に)
		\CGMODE.THUMB.NUM.SIZE(18)     // サムネイル画像閲覧枚数のフォントサイズ
		\CGMODE.THUMB.SIZE(160,120)    // サムネイル画像サイズ
		\CGMODE.THUMB.LIGHT(192,256)   // サムネイル画像の明るさ(通常時,マウスオーバー時)(0〜256)
		\CGMODE.CG.FADE(450,400,300)   // ＣＧフェードイン／クロスフェード／フェードアウト時間
		//\CGMODE.PAGE.FADE(450)         // (未)ページ切替フェード時間

		\CGMODE.PERNUM.XY(30,100)      // 達成率表示座標
		\CGMODE.PERNUM.SIZE(26)        // 数字１つ分の幅
		\CGMODE.PAGENUM.XY(338,450)    // 現在のページ表示座標
		\CGMODE.PAGENUM.SIZE(26)       // 数字１つ分の幅

	//---------------------------------------------------------------------
	// どの画像にどの機能を割り振るかの設定
	// 　ここで定義したボタン番号に対応する画像が、その機能となります。
	//   番号が大きいほうが前面に表示されます。
	//---------------------------------------------------------------------
		@es.CNUM.CGMODE.PLATE    = 101 //プレート(〜+ 上の\CGMODE.THUMB.NUMで定義した枚数分。デフォルトの場合だと8なので、つまり011〜018まで割り振られる事になる)
		@es.CNUM.CGMODE.PREVPAGE = 201 //前のページ
		@es.CNUM.CGMODE.NEXTPAGE = 202 //次のページ
		@es.CNUM.CGMODE.BACK     = 255 //戻る


Return[]
