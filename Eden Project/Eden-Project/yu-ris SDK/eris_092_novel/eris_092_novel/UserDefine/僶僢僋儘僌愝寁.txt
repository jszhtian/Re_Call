//============================================================================
//
// ERIS バックログ画面設計
//
//
//・ERIS ver0.69  004→005 変更点
//　　・ページ番号設定を追加
//
//・ERIS ver0.61  003→004 変更点
//　　・\LOG.SCROLL 命令を利用可能にした
//　　・\LOG.PAGE.NUM の制限の緩和
//
//・ERIS ver0.56  002→003 変更点
//　　・背景画像設定 \LOG.BACK.FILE の廃止
//　　・ボタン番号の自由化
//
//============================================================================
\VER.LOGDEF(005) //定義Version
//============================================================================

//-------------------------------------------------------------------------
//
// 全般
//
//-------------------------------------------------------------------------
#=es.SceneLog.Define


	//---------------------------------------------------------------------
	// レイヤセット設定
	//---------------------------------------------------------------------
		\LOG.LS.XY(48,80)                 // レイヤセット表示開始ＸＹ

	//---------------------------------------------------------------------
	// ボイス有りマーク画像設定
	//---------------------------------------------------------------------
		\LOG.VOICE("cg/backlog/btn_voice_off", "cg/backlog/btn_voice_over") // ファイル名
		\LOG.VOICE.XY(0,0)                // 表示位置設定(テキストレイヤ左上位置からの相対位置)

	//---------------------------------------------------------------------
	// ページ番号設定
	//---------------------------------------------------------------------
		\LOG.PAGENUM(0)                    // ページ番号表示する(=1)／しない(=0)
		\LOG.PAGENUM.XY(680,340)           // 現在のページ表示座標
		\LOG.PAGENUM.SIZE(14)              // 数字１つ分の幅
		\LOG.NUM.FILE("cg/backlog/num/num_")      // 数字画像ファイル名(末尾に0〜9の数字が付いたファイル名を読みに行きます)
		\LOG.SLH.FILE("cg/backlog/num/num_slash") // スラッシュ画像ファイル名

	//---------------------------------------------------------------------
	// テキストレイヤ設定
	//---------------------------------------------------------------------
		\LOG.TL.XY(30,600,500)            // 表示開始Ｘ座標、レイヤサイズＸＹ(ログ１個分)
		\LOG.TL.SPACE.Y(500)              // ログレイヤとログレイヤの行間隔
		                                  // (※ログレイヤのＹサイズ含む。４の倍数指定)

	//---------------------------------------------------------------------
	// テキスト設定
	//---------------------------------------------------------------------
		\LOG.TX.FONT.DEF("ＭＳ ゴシック") // デフォルトフォント
		\LOG.TX.SIZE(24)                  // テキストサイズ
		\LOG.TX.SPACE.XY(0,2)             // 文字間隔および行間隔
		\LOG.TX.COLOR(0xFFFFFF)           // テキストの標準色(RGB指定)

	//---------------------------------------------------------------------
	// 各種設定
	//---------------------------------------------------------------------
		\LOG.PAGE.NUM(1)                  // １画面にログを何頁分表示するか(1〜12)
		\LOG.PAGE.MAX(100)                // バックログを読み返せる頁数(〜500)
		\LOG.SAVE(1)                      // バックログをセーブデータに保存する
		\LOG.SCROLL(1)                    // テキストスクロールアニメーション数

	//---------------------------------------------------------------------
	// どの画像にどの機能を割り振るかの設定
	// 　ここで定義したボタン番号の画像が、その機能となります。
	//   番号が大きいほうが前面に表示されます。
	//---------------------------------------------------------------------
		@es.CNUM.LOG.BB   = 11 //■高速ログ↑スクロール
		@es.CNUM.LOG.B    = 12 //■ログ↑スクロール
		@es.CNUM.LOG.F    = 13 //■ログ↓スクロール
		@es.CNUM.LOG.FF   = 14 //■高速ログ↓スクロール
		@es.CNUM.LOG.BACK = 99 //■メイン画面に戻る


Return[]


//-------------------------------------------------------------------------
//
// 画像
//
//-------------------------------------------------------------------------
#=es.SceneLog.BtnSet


	//---------------------------------------------------------------------
	//
	// ボタン
	//
	// \BTN.CG( ボタンOFF時の画像, マウスが乗った時の画像 )
	// \BTN.XY( 表示Ｘ座標, 表示Ｙ座標 )
	// \BTN.MAP.A( α値のしきい値(この値以上の画像部分がボタンとして反応します) )
	// \BTN.KEY( ショートカットキー{一回の押下で一回のみ反応} ) 
	// \BTN.KEY.S( ショートカットキー{連続反応} ) 
	// \BTN.SE( オンマウス時ＳＥ, クリック時ＳＥ ) ※ファイル名orＳＥ定義文字列
	//
	// \BTNDEF ← 最後にBTNDEFでボタンを登録します。
	//
	//---------------------------------------------------------------------

		//■背景
		\BTN.CG("cg/backlog/back")
		\BTN.XY(000, 000)
		\BTNDEF(01) //ボタン登録

		//■高速ログ↑スクロール
		\BTN.CG("cg/backlog/btn_bb_off", "cg/backlog/btn_bb_over")
		\BTN.XY(730, 404)
		\BTN.KEY.S("PAGEUP")
		\BTNDEF(11) //ボタン登録

		//■ログ↑スクロール
		\BTN.CG("cg/backlog/btn_b_off", "cg/backlog/btn_b_over")
		\BTN.XY(730, 434)
		\BTN.KEY.S("UP", "MOUSE_U")
		\BTNDEF(12) //ボタン登録

		//■ログ↓スクロール
		\BTN.CG("cg/backlog/btn_f_off", "cg/backlog/btn_f_over")
		\BTN.XY(730, 464)
		\BTN.KEY.S("DOWN", "MOUSE_D")
		\BTNDEF(13) //ボタン登録

		//■高速ログ↓スクロール
		\BTN.CG("cg/backlog/btn_ff_off", "cg/backlog/btn_ff_over")
		\BTN.XY(730, 494)
		\BTN.KEY.S("PAGEDOWN")
		\BTNDEF(14) //ボタン登録

		//■メイン画面に戻る
		\BTN.CG("cg/backlog/btn_back_off", "cg/backlog/btn_back_over")
		\BTN.XY(680, 554)
		\BTN.KEY("ESC", "BS", "SPACE", "MOUSE_R")
		\BTNDEF(99) //ボタン登録


Return[]
