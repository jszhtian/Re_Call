\VER.FACEDEF(001) //定義Version
//============================================================================

#=フェイス画像定義


	//---------------------------------------------------------------------
	// フェイス画像のあるパス
	//---------------------------------------------------------------------
		\FACE.PATH("cg/face/")


	//---------------------------------------------------------------------
	// 一行喋ったら自動的にフェイス画像を消去するか(1=する/0=しない)
	//---------------------------------------------------------------------
		\FACE.AC.DEF(1)


	//---------------------------------------------------------------------
	// キャラ毎にフォルダを分けるか否か(1=する/0=しない)
	//---------------------------------------------------------------------
		\FACE.PATH.CHARAFLAG(1)


	//---------------------------------------------------------------------
	// キャラ毎にフォルダを分ける際のフォルダ名
	// \FACE.PATH.CHARA(キャラ番号, フォルダ名)
	//
	//                   ↓※この番号は macro.txt ファイルで定義した
	//                   ↓  「フェイスマクロ定義」のキャラ番号と
	//                   ↓   合わせてください。
	//---------------------------------------------------------------------
	    \FACE.PATH.CHARA(01, "yuuri/")
	    \FACE.PATH.CHARA(02, "eri/")
	    \FACE.PATH.CHARA(03, "ryo/")
	    \FACE.PATH.CHARA(04, "hiro/")


	//---------------------------------------------------------------------
	// キャラ毎の基準表示位置
	// \FACE.CHARA.XY(キャラ番号, Ｘ座標, Ｙ座標)
	//
	//                 ↓※この番号は macro.txt ファイルで定義した
	//                 ↓  「フェイスマクロ定義」のキャラ番号と
	//                 ↓   合わせてください。
	//---------------------------------------------------------------------
	    \FACE.CHARA.XY(01, 020, 346) //01:優理
	    \FACE.CHARA.XY(02, 010, 358) //02:枝理
	    \FACE.CHARA.XY(03, 010, 354) //03:良介
	    \FACE.CHARA.XY(04, 010, 354) //04:広伸


Return[]
