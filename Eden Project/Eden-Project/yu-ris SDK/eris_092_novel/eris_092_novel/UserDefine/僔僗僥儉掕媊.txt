\VER.SYSTEMDEF(007) //定義Version

#=es.System.Define


	//---------------------------------------------------------------------
	// デバッグモード時の設定
	// 　※ON/OFF自体の設定は projectconfig.txt 内でおこないます
	//---------------------------------------------------------------------

		//---------------------------------------------------------------------
		// デバッグモード時、キャプション名に付加する文字列
		//---------------------------------------------------------------------
			\DBG.CAPTION("  [デバッグモード]")

		//---------------------------------------------------------------------
		// デバッグキー設定
		// 　※キー割り当て用文字列詳細はERISマニュアルをご覧下さい。
		//---------------------------------------------------------------------
			\DBG.KEY.WINDOW(F12)                // デバッグウィンドウ表示キー
			\DBG.KEY.OPENSCRIPT(F11)            // スクリプトファイル表示キー
			\DBG.KEY.RESTART(F5)                // 高速再起動キー

		//---------------------------------------------------------------------
		// デバッグウィンドウ設定
		//---------------------------------------------------------------------
			\DBG.WINDOW.XY(0,0)       //デバッグウィンドウ表示位置
			\DBG.WINDOW.SIZE(600,128) //デバッグウィンドウサイズ


	//-------------------------------------------------------------------------
	// 起動時のスクリーンモード
	// 　0 = 必ず通常スクリーン
	// 　1 = 必ずフルスクリーン
	// 　2 = 初回起動時は通常スクリーン。次回起動時は前回終了時の設定
	// 　3 = 初回起動時はフルスクリーン。次回起動時は前回終了時の設定
	//-------------------------------------------------------------------------
		\SYS.SCREEN(2)


	//-------------------------------------------------------------------------
	// スクリーンセーバー起動防止設定
	// 　0 = スクリーンセーバーの起動を防止しない
	// 　1 = ウィンドウがアクティブの時はスクリーンセーバーの起動を防止する
	//-------------------------------------------------------------------------
		\SYS.SCREENSAVER(1)


	//---------------------------------------------------------------------
	// 起動チェック設定
	//---------------------------------------------------------------------
		\SYS.REQ.COLOR(1)        // 65536色未満の環境で起動させる(=0)
		\SYS.REQ.SOUND(1)        // サウンド機能が無い環境でも起動させる(=0)
		\SYS.REQ.READONLY(1)     // 読取専用環境で起動させる(=0)


return[]
