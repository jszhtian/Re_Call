#=ＢＧＭ定義


	//---------------------------------------------------------------------
	// ＢＧＭ素材のあるパス
	//---------------------------------------------------------------------
		\BGM.PATH("bgm/")


	//---------------------------------------------------------------------
	// ＢＧＭボリューム初期値(初回起動時にセットされる)
	//---------------------------------------------------------------------
		\BGM.VOL.DEF(224)  // 0〜256


	//---------------------------------------------------------------------
	// 全ＢＧＭのデフォルトフェード時間(ミリ秒)
	//---------------------------------------------------------------------
		\BGM.FADE.DEF(2000)


	//---------------------------------------------------------------------
	// ＢＧＭ簡易利用文字列定義
	//---------------------------------------------------------------------

		//例：「日常曲Ａ」という文字列を定義する
		//
		// ここで「日常曲Ａ」という文字列で設定を定義しておくと、
		// スクリプト中に \BGM("日常曲Ａ") という記述があった場合、
		// 以下に設定したパラメータが自動で設定された状態で再生されます。
		//
/*
		\BGM.FILE("bgm01")     // ファイル名
		\BGM.FADE(2000)        // フェード時間
		\BGM.POINT(-1,-1)      // 再生開始／終了／再開始地点
		\BGM.LOOP(0)           // 0=無限ループ再生, 1〜n=ループ回数
		\BGMDEF("日常曲Ａ")    // 登録.
*/

		//↓...現状100個分まで追加できます(β版仕様)



Return[]
