	//  !y → YY(年・西暦２桁)
	//  !M → MM(月・２桁)
	//  !D → DD(日・２桁)
	//  !W → WW(曜日)
	//  !G → GG(午前,午後)
	//  !H → HH(時・0〜23)

		$T(11,50)="金"                    // 金曜日, FRI
		$T(11,51)="土"                    // 土曜日, SAT

	//---------------------------------------------------------------------

		$T(11,53)="PM"                    // 午後, PM, p.m.
