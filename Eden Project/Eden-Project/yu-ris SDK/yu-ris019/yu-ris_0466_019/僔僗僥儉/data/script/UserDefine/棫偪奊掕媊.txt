#=ES.ST.DEFINE


	//---------------------------------------------------------------------
	// 立ち絵のデフォルト切替時間(ミリ秒)
	//   \キャラ名() マクロを使用した場合のデフォルト時間です
	//---------------------------------------------------------------------
	@T(28,01)=250


	//---------------------------------------------------------------------
	// 立ち絵簡易マクロを使用して複数立ち絵を消去しようとした際に、
	//   立ち絵が同時に消えない不具合をなくすか否か (1=なくす/0=なくさない)
	//
	// ※すでに以前のバージョンでスクリプトを途中まで組んでいる方は、
	// この設定により立ち絵の演出が変化してしまう可能性があります。
	//---------------------------------------------------------------------
	@T(28,91)=1


	//---------------------------------------------------------------------
	// 立ち絵フォルダ自動振り分け定義
	// 
	// 例えば
	// \SP.ST.FOLDER.DEF("yuu",,"yuu/")
	// と定義すると、
	// 先頭文字が「yuu」から始まる立ち絵ファイル名を指定した場合、
	// 自動的に「yuu/」フォルダ内の画像を探索します。
	//---------------------------------------------------------------------
	//フォーマット
	//\SP.ST.DEF( 画像ファイル先頭文字 , 画像ファイル末尾文字 ，対応するフォルダ名 )
	//
	\SP.ST.FOLDER.DEF("yuu",,"yuu/") //優理
	\SP.ST.FOLDER.DEF("eri",,"eri/") //枝理



	//---------------------------------------------------------------------
	// 立ち絵合成
	//
	// フォーマット
	// \SPC.ST.DEF( 差分画像ファイル先頭文字 , 差分画像ファイル末尾文字 ，ベース画像ファイル , 差分合成座標Ｘ, 差分合成座標Ｙ, 合成方式 )
	//
	// ・合成方式は、
	//    0…ベース画像の上に差分画像を乗せて合成する方式
	//    1…ベース画像から一部分を切り取り、そこに差分画像をはめ込む方式
	//   となっており、省略した場合は 0 となります。
	//---------------------------------------------------------------------
/*
	//■===== 優理
//	\SPC.ST.DEF("yuu_1a02" ,, "yuu_1a01", 129, 167, 0 ) //
	\SPC.ST.DEF("yuu_1a"   ,, "yuu_1a01" ) //
	\SPC.ST.DEF("yuu_1b"   ,, "yuu_1b01" ) //

	//■===== 枝理
	\SPC.ST.DEF("eri_1a"   ,, "eri_1a01" ) //
	\SPC.ST.DEF("eri_1b"   ,, "eri_1b01" ) //
*/


	//-------------------------------------------------------------------------
	// 立ち絵定期的アニメーション(目パチ、耳パタ等)
	//
	// フォーマット
	// \SPA.DEF( スクリプト上での記述文字列(先頭文字列) ，スクリプト上での記述文字列(末尾文字列) ，目パチファイル名, 座標Ｘ, 座標Ｙ, 枚数(最大4枚), 1コマあたりの速度(ミリ秒), 合成方式 )
	//
	// ・スクリプト上での記述文字列は、何文字でもかまいません。
	// ・ファイル名の末尾に 01 〜 04 の数字が付いたファイル名を読みに行きます。
	// ・座標Ｘ、座標Ｙは立ち絵画像からの相対座標になります。
	// ・合成方式は、
	//    0…ベース画像の上に差分画像を乗せて合成する方式
	//    1…ベース画像から一部分を切り取り、そこに差分画像をはめ込む方式
	//   となっており、省略した場合は 1 となります。
	//-------------------------------------------------------------------------
	\SPA.DEF("yuu_1a01",, "yuu_1a01_", 0, 0, 1, 60, 1)



RETURN[]
