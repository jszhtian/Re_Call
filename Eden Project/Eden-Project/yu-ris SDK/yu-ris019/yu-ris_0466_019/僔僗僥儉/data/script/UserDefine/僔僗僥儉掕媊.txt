#=ES.SYSTEM.DEFINE


	//-------------------------------------------------------------------------
	// 初回起動時のスクリーンモード
	// 　0 = 初回起動時は通常スクリーン (推奨)
	// 　1 = 初回起動時はフルスクリーン
	// 　2 = 0と同じ (以前との互換性の為)
	// 　3 = 1と同じ (以前との互換性の為)
	//-------------------------------------------------------------------------
		@T(17,01)=0


	//-------------------------------------------------------------------------
	// 非アクティブ時でも動作させる
	// 　1 = ON
	// 　0 = OFF
	//-------------------------------------------------------------------------
		@T(17,07)=1


	//-------------------------------------------------------------------------
	// ウィンドウを常に手前に表示する設定
	// 　1 = ON
	// 　0 = OFF
	//-------------------------------------------------------------------------
		@T(17,08)=0


	//-------------------------------------------------------------------------
	// スクリーンセーバー起動防止設定
	// 　1 = ウィンドウがアクティブの時はスクリーンセーバーの起動を防止する
	// 　0 = スクリーンセーバーの起動を防止しない
	//-------------------------------------------------------------------------
		@T(17,02)=1


	//---------------------------------------------------------------------
	// 起動チェック設定
	//---------------------------------------------------------------------
		@T(17,03)=1         // 65536色未満の環境で起動させる(=0)
		@T(17,04)=0         // サウンド機能が無い環境でも起動させる(=0)
		@T(17,05)=1         // 読取専用環境で起動させる(=0)


	//---------------------------------------------------------------------
	// スプライトの拡大縮小時の画質
	//---------------------------------------------------------------------
		@T(17,06)=1         // 0=低画質/1=高画質


	//---------------------------------------------------------------------
	// ムービー MEPG-1コーデックチェック
	//
	//   デバッグモード時、MPEG-1コーデック以外のムービーを再生しようと
	//   した場合にメッセージを表示します。
	//   意図的に MPEG-1 以外のムービーを使用したい場合を除いて、
	//   OFF にはしないでください。
	//   YMV ムービーとは無関係ですので、YMVムービーを使うために、
	//   OFF にする必要はありません。
	//---------------------------------------------------------------------
		@T(17,10)=1         // 1=ON/0=OFF


RETURN[]
