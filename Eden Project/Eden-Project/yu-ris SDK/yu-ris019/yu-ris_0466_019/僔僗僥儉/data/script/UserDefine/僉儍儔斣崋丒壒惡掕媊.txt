#=ES.VOICE.DEFINE


	//---------------------------------------------------------------------
	// 自動フォルダ分け・文字列定義
	//
	//   ボイスファイル名の先頭に「文字列」が存在した場合、
	//   自動的に「ボイスフォルダ名」のフォルダ内を探しに行きます。
	//   またキャラボイス番号の登録も兼ねているので、ここで登録した番号が
	//   コンフィグ設定でのキャラ別ボイス音量の順番に反映されます。
	//
	// [書き方]
	//   \VO.PATH.STR( 文字列, 文字列２, ... ) ※文字列４まで
	//   \VO.PATH.FOLDER( ボイスフォルダ名 )
	//   \VO.PATHDEF( キャラ番号 )
	//
	// [例]
	//   \VO.PATH.STR("eri_")
	//   \VO.PATH.FOLDER("枝理/")
	//   \VO.PATHDEF(02)
	//
	//   と定義しておけば、
	//
	//   \VO(eri_0001)
	//
	//   とボイス命令があった場合、
	//   通常は「voice/eri_0001(.wav/.ogg)」ファイルを探しに行くところを、
	//   「voice/枝理/eri_0001(.wav/.ogg)」ファイルを探しに行くようになります。
	//
	//---------------------------------------------------------------------
	//■01:優理
		\VO.PATH.STR("YUU_")		//ボイスファイル名先頭(※大文字小文字区別するので注意)
		\VO.PATH.FOLDER("YUU/")		//ボイスフォルダ名
		\VO.PATHDEF(01)				//キャラ番号登録

	//■02:枝理
		\VO.PATH.STR("ERI_")		//ボイスファイル名先頭(※大文字小文字区別するので注意)
		\VO.PATH.FOLDER("ERI/")		//ボイスフォルダ名
		\VO.PATHDEF(02)				//キャラ番号登録


	//---------------------------------------------------------------------
	// その他ボイスのキャラボイス番号設定
	//
	// 上で登録しなかったボイスファイル名の先頭が指定された場合、
	// 何番のキャラボイスとして音声再生するか設定します。
	//---------------------------------------------------------------------
		\VO.OTHER.PATH.FOLDER("")   //ボイスフォルダ名
		\VO.OTHER.PATHDEF(03)


RETURN[]
