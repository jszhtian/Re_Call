// キャラ名定義

#=ES.CHARNAME.DEFINE


	//---------------------------------------------------------------------
	// ■台詞として定義する記号
	//   名前の後に以下に定義した記号が続くと、
	//   その行はキャラの台詞として判断されます。
	//---------------------------------------------------------------------
		\CHAR.NAME.MARK("「")
		\CHAR.NAME.MARK("『")
		\CHAR.NAME.MARK("（")


	//---------------------------------------------------------------------
	// ■ウィンドウ表示対象にする名前の定義
	//
	// キャラの台詞のうち、ここに登録したキャラの名前は、
	// 名前ウィンドウに表示されるようになります。
	//
	// [書き方]
	// \CHAR.NAME( スクリプト上の文字 , 名前ウィンドウに表示される文字 , 名前ウィンドウに表示する画像ファイル, 名前ウィンドウに表示するグローバル文字列変数番号, 名前ウィンドウに表示するローカル文字列変数番号)
	//
	// ※第３パラメータに画像ファイルを指定すると、そちらが優先されて表示されます。
	//   (今後若干記述仕様変更の可能性もあります)
	//
	// ※第４パラメータに文字列変数番号を指定すると、対応した $G(変数番号) の内容が、
	//   第５パラメータに文字列変数番号を指定すると、対応した $L(変数番号) の内容が、
	//   「名前ウィンドウに表示される文字」として表示されます。
	//   例えば、
	//       \CHAR.NAME("主人公", "", "", 123)
	//   と定義しておけば、スクリプト中に
	//       主人公「〜」
	//   という記述が見つかった場合、「主人公」の部分が変数 $G(123) の中身に
	//   自動的に置き換えられます。
	//   (暫定仕様につき今後仕様変更されます)
	//
	//---------------------------------------------------------------------
		\CHAR.NAME("優理"			, "優理"			, "") //
		\CHAR.NAME("枝理"			, "枝理"			, "") //
		\CHAR.NAME("きゅーたろー"	, "きゅーたろー"	, "") //
		\CHAR.NAME("恵一"			, "恵一"			, "") //

		//※↓このように記述すれば、名前文字列の変わりに名前画像を表示することもできます。
	//	\CHAR.NAME("優理" , "" , "c01")
	//	\CHAR.NAME("枝理" , "" , "c02")

		//※↓このように記述すれば、名前文字列の変わりに文字列変数 $G(123) の中身を
		//    表示することもできます。
	//	\CHAR.NAME("主人公" , "" , "" , 123)




RETURN[]
